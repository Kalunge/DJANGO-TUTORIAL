Set Up Your Development Environment

python3 -m venv venv

ACTIVATE YOUR DEVELOPMENT ENVIRONMENT

source venv/bin/activate

INSTALL DJANGO USING PIP

pip install Django

CREATE A DJANGO PROJECT

django-admin startproject projectname

EDIT PROJECT STRUCTURE

$ mv projectname/manage.py ./

$ mv projectname/projectname/* projectname

$ rm -r projectname/projectname

START THE SERVER

python manage.py runserver

CREATE A DJANGO APPLICATION

python manage.py startapp hello_world

INSTALL YOUR APP IN SETTINGS.PY

INSTALLED_APPS = [ 'django.contrib.admin',
'django.contrib.auth',
'django.contrib.contenttypes',

CREATE A VIEW IN YOUR APP DIRECTORY

from django.shortcuts import render
def hello_world(request):
return render(request, 'hello_world.html', {})

create templates and html file in your app directory

$ mkdir hello_world/templates/
$ touch hello_world/templates/hello_world.html

MAP YOUR URL TO VISIT YOUR PAGE

from django.contrib import admin from django.urls import path, include urlpatterns = [
path('admin/', admin.site.urls),
path('', include('hello_world.urls')),
]

CREATE URLS.PY IN YOUR APP

touch hello_world/urls.py

from django.urls import path
from hello_world import views
urlpatterns = [
path('', views.hello_world, name='hello_world'),
]

runserver

python manage.py runserver
